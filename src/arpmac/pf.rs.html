1//! Packet Filter (PF) block list persistence and reload.
2//!
3//! arpmac blocks clients by loading rules into a dynamic PF **anchor** named
4//! `arpmac/block`. This avoids editing system `/etc/pf.conf`.
5//!
6//! ## Rule file format (`~/.config/arpmac/blocks.pf`)
7//!
8//! ```text
9//! block drop quick from 192.168.2.5 to any
10//! block drop quick from any to 192.168.2.5
11//! ```
12//!
13//! Each blocked IP gets two rules (egress + ingress).
14//!
15//! ## Reload command
16//!
17//! ```text
18//! sudo -S pfctl -a arpmac/block -f ~/.config/arpmac/blocks.pf
19//! ```
20//!
21//! When the block list is empty, the anchor is flushed instead:
22//!
23//! ```text
24//! sudo -S pfctl -a arpmac/block -F all
25//! ```
26
27use std::fs;
28use std::io;
29use std::path::PathBuf;
30
31mod dirs {
32    use std::path::PathBuf;
33
34    /// Return `~/.config/arpmac`, falling back to `/tmp/arpmac`.
35    pub fn config_dir() -> PathBuf {
36        std::env::var_os("HOME")
37            .map(|h| PathBuf::from(h).join(".config").join("arpmac"))
38            .unwrap_or_else(|| PathBuf::from("/tmp/arpmac"))
39    }
40}
41
42/// Absolute path to the generated PF rules file.
43pub fn blocks_path() -> PathBuf {
44    dirs::config_dir().join("blocks.pf")
45}
46
47/// Read blocked IPv4 addresses from the on-disk rules file.
48///
49/// Parses only outbound rules (`block drop quick from <ip> to any`),
50/// ignoring reverse rules that start with `any`.
51pub fn load_blocked() -> io::Result<Vec<String>> {
52    let path = blocks_path();
53    if !path.exists() {
54        return Ok(vec![]);
55    }
56    let text = fs::read_to_string(path)?;
57    Ok(text
58        .lines()
59        .filter_map(|line| {
60            let line = line.trim();
61            let rest = line.strip_prefix("block drop quick from ")?;
62            let ip = rest.split_whitespace().next()?;
63            if ip == "any" {
64                return None;
65            }
66            Some(ip.to_string())
67        })
68        .collect())
69}
70
71/// Rewrite `blocks.pf` with rules for every IP in `ips`.
72fn write_rules(ips: &[&str]) -> io::Result<()> {
73    let dir = dirs::config_dir();
74    fs::create_dir_all(&dir)?;
75    let mut rules = String::new();
76    for ip in ips {
77        rules.push_str(&format!("block drop quick from {ip} to any\n"));
78        rules.push_str(&format!("block drop quick from any to {ip}\n"));
79    }
80    fs::write(blocks_path(), rules)
81}
82
83/// Add `ip` to the block list and reload the PF anchor.
84///
85/// Requires a valid administrator password for [`crate::sudo::run`].
86pub fn block(ip: &str, password: &str) -> Result<(), String> {
87    let mut ips = load_blocked().map_err(|e| e.to_string())?;
88    if ips.iter().any(|x| x == ip) {
89        return reload(password).map_err(|e| e.to_string());
90    }
91    ips.push(ip.to_string());
92    let refs: Vec<&str> = ips.iter().map(String::as_str).collect();
93    write_rules(&refs).map_err(|e| e.to_string())?;
94    reload(password).map_err(|e| e.to_string())
95}
96
97/// Remove `ip` from the block list and reload the PF anchor.
98pub fn unblock(ip: &str, password: &str) -> Result<(), String> {
99    let ips = load_blocked().map_err(|e| e.to_string())?;
100    let refs: Vec<&str> = ips
101        .iter()
102        .filter(|x| *x != ip)
103        .map(String::as_str)
104        .collect();
105    write_rules(&refs).map_err(|e| e.to_string())?;
106    reload(password).map_err(|e| e.to_string())
107}
108
109/// Apply the current `blocks.pf` to the running PF anchor via `pfctl`.
110fn reload(password: &str) -> io::Result<()> {
111    let ips = load_blocked()?;
112    if ips.is_empty() {
113        let out = crate::sudo::run(password, &["pfctl", "-a", "arpmac/block", "-F", "all"])?;
114        if out.status.success() {
115            return Ok(());
116        }
117        return Ok(());
118    }
119    let path = blocks_path().to_string_lossy().to_string();
120    let out = crate::sudo::run(password, &["pfctl", "-a", "arpmac/block", "-f", &path])?;
121    if out.status.success() {
122        return Ok(());
123    }
124    Err(io::Error::other(format!(
125        "{}",
126        String::from_utf8_lossy(&out.stderr)
127    )))
128}
