1//! Internet Sharing detection and Wi‑Fi client discovery.
2//!
3//! All network I/O goes through native macOS shell commands — no extra crates.
4//!
5//! ## Discovery strategy
6//!
7//! ```text
8//!   ┌─────────────────┐     live signal      ┌──────────────────┐
9//!   │ arp -a          │ ───────────────────► │ connected = true │
10//!   │ (on bridge100)  │                      │ label: ● now     │
11//!   └─────────────────┘                      └──────────────────┘
12//!
13//!   ┌─────────────────┐     historical       ┌──────────────────┐
14//!   │ /var/db/        │ ───────────────────► │ connected = false│
15//!   │ dhcpd_leases    │   (names, old IPs)   │ label: past      │
16//!   └─────────────────┘                      └──────────────────┘
17//!
18//!   Secondary liveness: bridge ARP, nettop byte counters (no per-host ping)
19//! ```
20//!
21//! a standard /var/db/dhcpd_leases file looks like this:
22//!
23//! ```text
24//! {
25//!     ip_address=192.168.64.2
26//!     hw_address=1,ee:df:e4:7b:50:4d
27//!     identifier=1,ee:df:e4:7b:50:4d
28//!     lease=0x6a02282f
29//!     name=minint-1llret8
30//! }
31//! ```
32//!
33//! See the [crate-level guide](crate) for terminology and data-flow diagrams.
34

41/// Path to macOS DHCP lease database written by `bootpd`.
42const LEASES: &str = "/var/db/dhcpd_leases";

44/// Result of checking whether Internet Sharing is active.
45#[derive(Clone, Debug, Default)]
46pub struct Sharing {
47    /// `true` when a bridge interface with an IPv4 address exists.
48    pub active: bool,
49    /// BSD interface name, usually `bridge100`.
50    pub interface: String,
51    /// Gateway IP on the shared subnet, usually `192.168.2.1`.
52    pub gateway: String,
53    /// CIDR prefix length derived from netmask, usually `24`.
54    pub prefix: u8,
55}
56
57/// One row in the client table.
58#[derive(Clone, Debug)]
59pub struct Client {
60    /// IPv4 address on the shared subnet.
61    pub ip: String,
62    /// Hardware MAC address.
63    pub mac: String,
64    /// Hostname from DHCP lease, if reported by the client.
65    pub name: String,
66    /// On the sharing bridge right now (bridge ARP or active nettop flows).
67    pub connected: bool,
68    /// Whether this IP is in the PF block list.
69    pub blocked: bool,
70    /// Bytes received from this client (nettop delta, current refresh).
71    pub bytes_in: u64,
72    /// Bytes sent to this client (nettop delta, current refresh).
73    pub bytes_out: u64,
74}

76/// Detect Internet Sharing by looking for a bridge interface with an IPv4 address.
77///
78/// Tries `bridge100` first, then any `bridge*` from `ifconfig -l`.
79///
80/// # Example output when sharing is off
81///
82/// ```text
83/// Sharing { active: false, .. Default }
84/// ```
85pub fn detect_sharing() -> Sharing {
86    let mut sharing = Sharing::default();
87    if let Some((iface, gw, prefix)) = find_bridge() {
88        sharing.active = true;
89        sharing.interface = iface;
90        sharing.gateway = gw;
91        sharing.prefix = prefix;
92    }
93    sharing
94}
95
96/// Locate the Internet Sharing bridge and parse its gateway IP + prefix.
97fn find_bridge() -> Option<(String, String, u8)> {
98    if let Some(info) = parse_ifconfig("bridge100") {
99        return Some(("bridge100".into(), info.0, info.1));
100    }
101    let list = cmd_output(&["ifconfig", "-l"]);
102    for iface in list.split_whitespace().filter(|i| i.starts_with("bridge")) {
103        if let Some(info) = parse_ifconfig(iface) {
104            return Some((iface.to_string(), info.0, info.1));
105        }
106    }
107    None
108}

110/// Parse `ifconfig <iface>` output for `inet` and `netmask` lines.
111///
112/// Returns `(gateway_ip, prefix_length)`.
113fn parse_ifconfig(iface: &str) -> Option<(String, u8)> {
114    let out = cmd_output(&["ifconfig", iface]);
115    if out.contains("does not exist") {
116        return None;
117    }
118    let mut ip = None;
119    let mut mask = None;
120    for line in out.lines() {
121        let line = line.trim();
122        if let Some(rest) = line.strip_prefix("inet ") {
123            let addr = rest.split_whitespace().next()?;
124            ip = Some(addr.to_string());
125        }
126        if let Some(rest) = line.strip_prefix("netmask ") {
127            let hex = rest.split_whitespace().next()?;
128            if let Some(prefix) = netmask_to_prefix(hex) {
129                mask = Some(prefix);
130            }
131        }
132    }
133    Some((ip?, mask.unwrap_or(24)))
134}

136/// Convert macOS hex netmask (e.g. `0xffffff00`) to CIDR prefix length.
137fn netmask_to_prefix(hex: &str) -> Option<u8> {
138    let hex = hex.strip_prefix("0x")?;
139    let n = u32::from_str_radix(hex, 16).ok()?;
140    Some(n.count_ones() as u8)
141}

143/// Build the full client list by merging bridge ARP, DHCP leases, and bandwidth data.
144///
145/// Returns empty vec when [`Sharing::active`] is false.
146///
147/// Sort order: connected clients first, then by IP ascending.
148/// Build the full client list: DHCP leases first, then bridge ARP, then nettop bandwidth.
149///
150/// Order is intentional for speed — `/var/db/dhcpd_leases` is a single small file;
151/// [`connected_on_bridge`] uses `arp -a -i bridge100` (scoped, not full `arp -a`).
152/// Per-host `ping` was removed — it ran sequentially and blocked for ~1 s per lease.
153pub fn discover_clients(sharing: &Sharing, blocked: &[String]) -> Vec<Client> {
154    if !sharing.active {
155        return vec![];
156    }
157
158    let mut map: HashMap<String, Client> = HashMap::new();
159
160    if let Ok(text) = fs::read_to_string(LEASES) {
161        for record in parse_leases(&text) {
162            if !in_subnet(&record.ip, &sharing.gateway, sharing.prefix) {
163                continue;
164            }
165            if record.ip == sharing.gateway {
166                continue;
167            }
168            map.insert(
169                record.ip.clone(),
170                Client {
171                    ip: record.ip,
172                    mac: record.mac,
173                    name: record.name,
174                    connected: false,
175                    blocked: false,
176                    bytes_in: 0,
177                    bytes_out: 0,
178                },
179            );
180        }
181    }
182
183    let online = connected_on_bridge(&sharing.gateway, sharing.prefix, &sharing.interface);
184    for (ip, mac) in &online {
185        map.entry(ip.clone())
186            .and_modify(|c| {
187                c.connected = true;
188                if c.mac.is_empty() {
189                    c.mac = mac.clone();
190                }
191            })
192            .or_insert_with(|| Client {
193                ip: ip.clone(),
194                mac: mac.clone(),
195                name: String::new(),
196                connected: true,
197                blocked: false,
198                bytes_in: 0,
199                bytes_out: 0,
200            });
201    }
202
203    let bandwidth = nettop_bandwidth(&sharing.gateway, sharing.prefix);
204    let mut clients: Vec<Client> = map.into_values().collect();
205    for c in &mut clients {
206        if online.contains_key(&c.ip) {
207            c.connected = true;
208        } else if bandwidth.contains_key(&c.ip) {
209            c.connected = true;
210        }
211        if let Some((in_b, out_b)) = bandwidth.get(&c.ip) {
212            c.bytes_in = *in_b;
213            c.bytes_out = *out_b;
214        }
215        c.blocked = blocked.iter().any(|b| b == &c.ip);
216    }
217    clients.sort_by(|a, b| b.connected.cmp(&a.connected).then_with(|| a.ip.cmp(&b.ip)));
218    clients
219}

221/// Parse `arp -a -i <iface>` for complete MAC entries on the sharing bridge.
222///
223/// Uses interface-scoped ARP instead of full `arp -a` (which scans every interface
224/// and can be slow when many stale entries exist on `en0`).
225fn connected_on_bridge(gateway: &str, prefix: u8, iface: &str) -> HashMap<String, String> {
226    let out = cmd_output(&["arp", "-a", "-i", iface]);
227    let mut map = HashMap::new();
228    for line in out.lines() {
229        let Some(start) = line.find('(') else {
230            continue;
231        };
232        let Some(end) = line[start + 1..].find(')') else {
233            continue;
234        };
235        let ip = &line[start + 1..start + 1 + end];
236        if ip == gateway || !in_subnet(ip, gateway, prefix) {
237            continue;
238        }
239        let Some(at) = line.find(" at ") else {
240            continue;
241        };
242        let mac = line[at + 4..].split_whitespace().next().unwrap_or("");
243        if mac == "(incomplete)" || !mac.contains(':') {
244            continue;
245        }
246        map.insert(ip.to_string(), mac.to_string());
247    }
248    map
249}

251/// One record parsed from `/var/db/dhcpd_leases`.
252struct LeaseRecord {
253    ip: String,
254    mac: String,
255    name: String,
256}
257
258/// Parse the brace-delimited lease file format written by `bootpd`.
259///
260/// ```text
261/// {
262///     ip_address=192.168.64.2
263///     hw_address=1,ee:df:e4:7b:50:4d
264///     identifier=1,ee:df:e4:7b:50:4d
265///     lease=0x6a02282f
266///     name=minint-1llret8
267/// }
268/// ```
269fn parse_leases(text: &str) -> Vec<LeaseRecord> {
270    let mut records = vec![];
271    let mut ip = String::new();
272    let mut mac = String::new();
273    let mut name = String::new();
274
275    for line in text.lines() {
276        let line = line.trim();
277        if line == "{" {
278            ip.clear();
279            mac.clear();
280            name.clear();
281        } else if line == "}" {
282            if !ip.is_empty() {
283                records.push(LeaseRecord {
284                    ip: ip.clone(),
285                    mac: mac.clone(),
286                    name: name.clone(),
287                });
288            }
289        } else if let Some(v) = line.strip_prefix("ip_address=") {
290            ip = v.to_string();
291        } else if let Some(v) = line.strip_prefix("hw_address=") {
292            mac = v.strip_prefix("1,").unwrap_or(v).to_string();
293        } else if let Some(v) = line.strip_prefix("name=") {
294            name = v.to_string();
295        }
296    }
297    records
298}

300/// Return `true` if `ip` is in the same IPv4 subnet as `gateway`/`prefix`.
301fn in_subnet(ip: &str, gateway: &str, prefix: u8) -> bool {
302    let Ok(ip) = Ipv4Addr::from_str(ip) else {
303        return false;
304    };
305    let Ok(gw) = Ipv4Addr::from_str(gateway) else {
306        return false;
307    };
308    let mask = if prefix == 0 {
309        0
310    } else {
311        u32::MAX << (32 - prefix)
312    };
313    (u32::from(ip) & mask) == (u32::from(gw) & mask)
314}

316/// Run `nettop` once and aggregate byte counters per client IP on the shared subnet.
317///
318/// Command: `nettop -L 1 -n -x -d -t external`
319fn nettop_bandwidth(gateway: &str, prefix: u8) -> HashMap<String, (u64, u64)> {
320    let out = cmd_output(&["nettop", "-L", "1", "-n", "-x", "-d", "-t", "external"]);
321    let mut map: HashMap<String, (u64, u64)> = HashMap::new();
322    for line in out.lines().skip(1) {
323        let cols: Vec<&str> = line.split(',').collect();
324        if cols.len() < 6 {
325            continue;
326        }
327        let conn = cols[1];
328        if !(conn.starts_with("tcp") || conn.starts_with("udp")) {
329            continue;
330        }
331        let ip = extract_client_ip(conn, gateway, prefix);
332        let Some(ip) = ip else { continue };
333        let in_b = cols[4].parse::<u64>().unwrap_or(0);
334        let out_b = cols[5].parse::<u64>().unwrap_or(0);
335        let e = map.entry(ip).or_insert((0, 0));
336        e.0 += in_b;
337        e.1 += out_b;
338    }
339    map
340}

342/// Extract a client IP from a nettop connection field.
343///
344/// Example input: `tcp4 192.168.2.5:54321<->8.8.8.8:443`
345fn extract_client_ip(conn: &str, gateway: &str, prefix: u8) -> Option<String> {
346    let rest = conn.split_whitespace().nth(1)?;
347    for part in rest.split("<->") {
348        let addr = part.split(':').next()?;
349        if in_subnet(addr, gateway, prefix) && addr != gateway {
350            return Some(addr.to_string());
351        }
352    }
353    None
354}

356/// Look up the hardware vendor for a MAC address using api.macvendors.com.
357///
358/// Uses the first three octets (OUI). Requires network access.
359pub fn lookup_vendor(mac: &str) -> Result<String, String> {
360    let oui: String = mac.split(':').take(3).collect::<Vec<_>>().join(":");
361    if oui.len() < 8 {
362        return Err("invalid MAC".into());
363    }
364    let url = format!("https://api.macvendors.com/{oui}");
365    let out = Command::new("curl")
366        .args(["-sf", &url])
367        .output()
368        .map_err(|e| e.to_string())?;
369    if !out.status.success() {
370        return Err("lookup failed".into());
371    }
372    Ok(String::from_utf8_lossy(&out.stdout).trim().to_string())
373}

375/// Format a byte count for table display (`—`, `512`, `1.2K`, `3.4M`).
376pub fn format_bytes(n: u64) -> String {
377    const K: u64 = 1024;
378    if n >= K * K {
379        format!("{:.1}M", n as f64 / (K * K) as f64)
380    } else if n >= K {
381        format!("{:.1}K", n as f64 / K as f64)
382    } else if n > 0 {
383        format!("{n}")
384    } else {
385        "—".into()
386    }
387}

389/// Run a command and return stdout as UTF-8 (empty string on failure).
390fn cmd_output(args: &[&str]) -> String {
391    Command::new(args[0])
392        .args(&args[1..])
393        .output()
394        .map(|o| String::from_utf8_lossy(&o.stdout).into_owned())
395        .unwrap_or_default()
396}
